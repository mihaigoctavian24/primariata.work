```typescript
// Bucket: "cereri-documente"
// Path convention: ${primarieId}/admin/${folderPath}/${filename}
// The "admin" subfolder separates admin-uploaded docs from citizen cereri docs

// List files in a folder prefix:
const { data: files } = await supabase.storage
  .from("cereri-documente")
  .list(`${primarieId}/admin/`, {
    limit: 100,
    sortBy: { column: "updated_at", order: "desc" },
  });
// files: Array<{ name: string; id: string; updated_at: string; metadata: { size: number; mimetype: string } }>

// Signed URL (1 hour):
const { data: { signedUrl } } = await supabase.storage
  .from("cereri-documente")
  .createSignedUrl(`${primarieId}/admin/${filename}`, 3600);

// Upload:
await supabase.storage
  .from("cereri-documente")
  .upload(`${primarieId}/admin/${filename}`, file, { upsert: false });
```
