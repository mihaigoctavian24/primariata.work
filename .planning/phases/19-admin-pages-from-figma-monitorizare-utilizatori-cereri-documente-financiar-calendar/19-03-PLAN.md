<!-- SLA computation helper -->
```typescript
function computeSlaRemaining(dataTermen: string | null): number | null {
  if (!dataTermen) return null;
  const diff = new Date(dataTermen).getTime() - Date.now();
  return Math.ceil(diff / (1000 * 60 * 60 * 24)); // days remaining
}
// < 0 = overdue, 0-2 = critical, 3-7 = warning, > 7 = ok
```
